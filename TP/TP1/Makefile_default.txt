EXEC=tri

all: $(EXEC)

tri: utile.o tri.o main.o 
	gcc -o tri utile.o tri.o main.o

utile.o: utile.c
	gcc -o utile.o -c utile.c -W -Wall -ansi -pedantic

tri.o: tri.c utile.h
	gcc -o tri.o -c tri.c -W -Wall -ansi -pedantic

main.o: main.c utile.h tri.h
	gcc -o main.o -c main.c -W -Wall -ansi -pedantic

clean :
	rm *.o $(EXEC)
